        // the setup function runs once when you press reset or power the board
void setup() {
  // initialize three LEDs as an output.
  pinMode(11, OUTPUT); // first LED
  pinMode(8, OUTPUT); // second LED
  pinMode(5, OUTPUT); // third LED
}

// the loop function runs over and over again forever
void loop() {
  digitalWrite(11, HIGH);  // turn the first LED on (HIGH is the voltage level)
  delay(1000);                      // wait for a second
  digitalWrite(11, LOW);   // turn the first LED off by making the voltage LOW
  delay(1000);
  digitalWrite(8, HIGH);  // turn the second LED on (HIGH is the voltage level)
  delay(1000);                      // wait for a second
  digitalWrite(8, LOW);   // turn the second LED off by making the voltage LOW
  delay(1000);
  digitalWrite(5, HIGH);  // turn the third LED on (HIGH is the voltage level)
  delay(1000);                      // wait for a second
  digitalWrite(5, LOW);   // turn the third LED off by making the voltage LOW
  delay(1000);                    // wait for a second
}
